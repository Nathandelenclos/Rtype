95  if (message == "connection accepted") {
96  _socket->setInit(true);

99  if (component->getType() == ComponentType::TEXT) {

102  }
103  }
104  }

113  timerecv = *static_cast<timeval *>(p->data);

115

121  ", " + std::to_string(_pingTime.tv_usec % 1000) + "ms");
122  }
123  }
124  }
